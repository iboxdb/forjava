Project Description

iBoxDB is a fast transactional table style document NoSQL Application Database, easily store, process objects and documents, traditional table with unstructured data, zero configuration, pure JAVA and .NET engines, no dependencies.

iBoxDB has a well designed interface with great performance and capability for agile development. You can create applications with database services embedded and deploy it on mobiles, desktops, servers, to persist your data from anywhere.

                .ensureIndex("Table", Member.class, "Name"); 
  //Tree->Parts

                .ensureIndex("Node", Node.class, "Parent"); 

  db.getConfig().ensureTable("Part", Part.class, "Id") 
                .ensureIndex("Part", Part.class, "Node");

  DatabaseConfig cfg = db.getConfig().DBConfig;
  cfg.CacheLength = cfg.mb(512);
  cfg.FileIncSize = (int)cfg.mb(4);

  auto = db.open(); 
}

boolean exists(Box box) { 
  return box.selectCount("from Table limit 0,1") > 0; 
} 

<V> Iterable<V> startsWith(Box box, Class<V> vClass, String prefix) { 
  String begin = prefix; 
  char[] cs = prefix.toCharArray(); 
  cs[cs.length - 1] = (char) ((short) cs[cs.length - 1] + 1); 
  String end = new String(cs); 
  return box.select(vClass,"from Table where Name>=? & Name<?",begin,end); 
}

void treeJoin(Box box, long nodeRoot, int maxLevel, StringBuilder result) { 
  if (maxLevel > 5) { return; } 
  char[] cs = new char[maxLevel * 2]; Arrays.fill(cs, ' '); 
  for (Node node : box.select(Node.class, "from Node where Parent==?", nodeRoot)) { 
    result.append(cs).append("NODE:")
          .append("->").append(node.Id).append("\r\n"); 
    for (Part part : box.select(Part.class, "from Part where Node==?", node.Id)) { 
      result.append(cs).append(" PART:")
            .append("->").append(part.Id).append("\r\n"); 
    } 
    treeJoin(box, node.Id, maxLevel + 1, result); 
  } 
}

synchronized static void onApplicationClose() { 
  auto.getDatabase().close(); 
}
Use DatabaseServer to Manage Configs 
using iBoxDB.LocalServer;
using iBoxDB.LocalServer.IO;

public class ApplicationServer : LocalDatabaseServer 
{ 
  public static ApplicationServer App = new ApplicationServer(); 
  public static IDatabase DB(long address) 
  { 
    return App.GetInstance(address); 
  }

  abstract class Config : BoxFileStreamConfig 
  { 
    public Config() : base() 
    { 
      CacheLength = MB(512); 
      FileIncSize = (int)MB(4); 
      ReadStreamCount = 8; 
    } 
  } 
                
  class MainConfig : Config 
  { 
    public MainConfig() : base() 
    { 
      EnsureTable<Member>("Member", "Id"); 
      EnsureIndex<Member>("Member", "Name(20)");
      EnsureTable<Product>("Product", "Type", "UID"); 
    }
  }
  
  class InventoryConfig : Config 
  { 
    public InventoryConfig() : base() 
    { 
      EnsureTable<Item>("Item", "Id"); 
    } 
  } 
                
  protected override DatabaseConfig BuildDatabaseConfig(long address) 
  { 
    switch (address) 
    { 
      case 1L: return new MainConfig(); 
      case 2L: return new InventoryConfig(); 
      default: return null; 
    } 
  } 
} 
                
void Anywhere1() 
{ 
  var auto = ApplicationServer.DB(1L).Get(); 
  auto.Insert("Member", new Member()); 
} 
                
void Anywhere2() 
{ 
  var db = ApplicationServer.DB(2L); 
  using (var box = db.Cube()) 
  { 
    box["Item"].Insert(new Item()); 
    CommitResult cr = box.Commit(); 
  } 
}

void OnApplicationClose() 
{ 
  ApplicationServer.App.Dispose(); 
}
